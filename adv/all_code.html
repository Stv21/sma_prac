# ================================================== # 1. Content Analysis # ================================================== import pandas as pd import re from sklearn.feature_extraction.text import CountVectorizer from sklearn.decomposition import LatentDirichletAllocation df = pd.read_csv("https://raw.githubusercontent.com/Stv21/sma_prac/refs/heads/main/Tweets.csv") texts = df['text'].astype(str) def clean_text(text): text = text.lower() text = re.sub(r"http\S+", "", text) text = re.sub(r"@\w+", "", text) text = re.sub(r"[^a-z\s]", "", text) return text texts = texts.apply(clean_text) vectorizer = CountVectorizer(stop_words='english', max_features=20) X = vectorizer.fit_transform(texts) print("Top Keywords:") print(vectorizer.get_feature_names_out()) lda = LatentDirichletAllocation(n_components=3, random_state=42) lda.fit(X) print("\nTopics:") for i, topic in enumerate(lda.components_): print(f"Topic {i+1}:") print([vectorizer.get_feature_names_out()[j] for j in topic.argsort()[-5:]]) # ================================================== # 2. Social Network Analysis # ================================================== import networkx as nx import matplotlib.pyplot as plt from sklearn.cluster import KMeans from sklearn.manifold import SpectralEmbedding G = nx.karate_club_graph() def girvan_newman(G): G = G.copy() while len(list(nx.connected_components(G))) == 1: eb = nx.edge_betweenness_centrality(G) G.remove_edge(*max(eb, key=eb.get)) return list(nx.connected_components(G)) communities = girvan_newman(G) color_map = {} for i, c in enumerate(communities): for node in c: color_map[node] = i colors = [color_map[n] for n in G.nodes()] X = SpectralEmbedding(n_components=5).fit_transform(nx.to_numpy_array(G)) kmeans_labels = KMeans(n_clusters=2, n_init=10).fit_predict(X) plt.figure(figsize=(8,6)) nx.draw(G, node_color=colors, with_labels=True, cmap=plt.cm.Set1) plt.title("Girvan–Newman Community Detection (Colored)") plt.show()
